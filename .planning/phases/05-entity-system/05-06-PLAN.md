### 1. Entity Lifecycle Tests

Create `tests/entity-lifecycle.test.ts`:

```typescript
import { describe, it, expect, beforeEach, afterEach, vi } from 'vitest';
import { EntityBase, EntityEventEmitter } from '../src/entities/Entity.base';

describe('EntityBase', () => {
  class TestEntity extends EntityBase {
    async initialize(): Promise<void> {
      await super.initialize({ deltaTime: 0, gameState: {}, physics: {} });
    }
  }
  
  it('should start in created state', () => {
    const entity = new TestEntity('test', 'player', { x: 0, y: 0 });
    expect(entity.state).toBe('created');
  });
  
  it('should transition to ready on initialize', async () => {
    const entity = new TestEntity('test', 'player', { x: 0, y: 0 });
    await entity.initialize();
    expect(entity.state).toBe('ready');
  });
  
  it('should transition to active on first update', async () => {
    const entity = new TestEntity('test', 'player', { x: 0, y: 0 });
    await entity.initialize();
    entity.update(0.016, { deltaTime: 0.016, gameState: {}, physics: {} });
    expect(entity.state).toBe('active');
  });
  
  it('should emit spawn event on initialize', async () => {
    const entity = new TestEntity('test', 'player', { x: 0, y: 0 });
    const emitSpy = vi.spyOn(entity.events, 'emit');
    await entity.initialize();
    expect(emitSpy).toHaveBeenCalledWith(
      expect.objectContaining({ type: 'spawn' })
    );
  });
  
  it('should emit destroy event on destroy', async () => {
    const entity = new TestEntity('test', 'player', { x: 0, y: 0 });
    const emitSpy = vi.spyOn(entity.events, 'emit');
    await entity.destroy();
    expect(emitSpy).toHaveBeenCalledWith(
      expect.objectContaining({ type: 'destroy' })
    );
  });
});

describe('EntityEventEmitter', () => {
  it('should register and call listeners', () => {
    const emitter = new EntityEventEmitter();
    const handler = vi.fn();
    
    emitter.on('collision', handler);
    emitter.emit({ type: 'collision', entityId: 'test', timestamp: Date.now() });
    
    expect(handler).toHaveBeenCalledTimes(1);
  });
  
  it('should remove listeners', () => {
    const emitter = new EntityEventEmitter();
    const handler = vi.fn();
    
    emitter.on('collision', handler);
    emitter.off('collision', handler);
    emitter.emit({ type: 'collision', entityId: 'test', timestamp: Date.now() });
    
    expect(handler).not.toHaveBeenCalled();
  });
});
```

### 2. Game Flow Tests

Create `tests/game-flow.test.ts`:

```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import { useGameStore } from '../src/store';

describe('Game Flow', () => {
  beforeEach(() => {
    // Reset store state
    useGameStore.getState().$reset();
  });
  
  describe('1P Mode', () => {
    it('should transition from MENU to GAME_MODE_SELECT', () => {
      const store = useGameStore.getState();
      store.setGameState('MENU');
      store.enterGameModeSelect();
      expect(store.gameState).toBe('GAME_MODE_SELECT');
    });
    
    it('should start game with AI opponent', () => {
      const store = useGameStore.getState();
      store.setGameMode('1P');
      store.setDifficulty('normal');
      store.startGame();
      expect(store.gameState).toBe('PLAYING');
      expect(store.gameMode).toBe('1P');
    });
    
    it('should handle player death', () => {
      const store = useGameStore.getState();
      store.setGameState('PLAYING');
      store.updateWinner('player2');
      expect(store.gameState).toBe('GAME_OVER');
    });
  });
  
  describe('2P Mode', () => {
    it('should start game with two players', () => {
      const store = useGameStore.getState();
      store.setGameMode('2P');
      store.startGame();
      expect(store.gameState).toBe('PLAYING');
      expect(store.gameMode).toBe('2P');
    });
    
    it('should handle round restart', () => {
      const store = useGameStore.getState();
      store.setGameMode('2P');
      store.startGame();
      store.p1Score = 2;
      store.startRound();
      expect(store.gameState).toBe('PLAYING');
      // Scores should be reset or incremented based on rules
    });
  });
});
```

### 3. Player Entity Tests

Create `tests/player-entity.test.ts`:

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { Player } from '../src/entities/Player';

// Mock dependencies
vi.mock('../systems/PhysicsSystem', () => ({
  getPhysicsSystem: () => ({
    createBody: () => ({
      rigidBody: {
        translation: () => ({ x: 0, y: 4, z: 0 }),
        rotation: () => ({ x: 0, y: 0, z: 0, w: 1 }),
        applyImpulse: vi.fn(),
        setTranslation: vi.fn(),
        setLinvel: vi.fn(),
      },
      collider: {
        setFriction: vi.fn(),
      }
    }),
    destroyBody: vi.fn(),
  })
}));

describe('Player Entity', () => {
  const defaultInput = () => ({ forward: false, backward: false, left: false, right: false, boost: false });
  
  it('should create player with correct id and color', () => {
    const player = new Player('player1', 0xff4444, { x: 0, y: 4, z: 0 }, defaultInput);
    expect(player.id).toBe('player1');
    expect(player.color).toBe(0xff4444);
  });
  
  it('should update position from physics', () => {
    const player = new Player('player1', 0xff4444, { x: 0, y: 4, z: 0 }, defaultInput);
    // Physics mock returns position 0, 4, 0
    expect(player.position.x).toBe(0);
  });
  
  it('should handle death when y < -10', () => {
    const player = new Player('player1', 0xff4444, { x: 0, y: 4, z: 0 }, defaultInput);
    // Simulate falling below threshold
    player.rigidBody.translation = () => ({ x: 0, y: -15, z: 0 });
    // player.update called with death condition
    expect(player.isDead).toBe(true);
  });
});
```
