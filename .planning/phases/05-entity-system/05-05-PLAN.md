### 2. Integrate Entity Lifecycle

Modify resetEntities():

```typescript
async function resetEntities() {
  // Cleanup existing
  await player1?.destroy();
  await player2?.destroy();
  await arena?.destroy();
  // ... cleanup all entities
  
  const state = useGameStore.getState();
  const isOnePlayer = state.gameMode === '1P';
  
  // Create with lifecycle
  player1 = new Player('player1', 0xff4444, { x: -15, y: 4, z: 0 }, getPlayer1InputUnified);
  player2 = new Player('player2', 0x4444ff, { x: 15, y: 4, z: 0 }, 
      isOnePlayer ? () => aiController!.getInput() : getPlayer2InputUnified);
  arena = new Arena();
  particles = new ParticleSystem();
  lightning = new LightningSystem();
  shockwaves = new ShockwaveSystem();
  
  // Initialize all entities
  const context = { deltaTime: 0, gameState, physics: physicsSystem };
  await player1.initialize(context);
  await player2.initialize(context);
  await arena.initialize(context);
  // ... initialize others
}
```
