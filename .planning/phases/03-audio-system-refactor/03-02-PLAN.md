```typescript
/**

export interface PlaybackRequest {
  url: string;
  loop?: boolean;
  volume?: number;
}

/**
 * PlaybackEvent - Events emitted by AudioSystem
 * Discriminated union for type-safe event handling
 */
export type PlaybackEvent = 
  | { type: 'lifecycle-changed'; state: AudioLifecycle }
  | { type: 'playback-started'; url: string }
  | { type: 'playback-stopped' }
  | { type: 'error'; message: string };

/**
 * Event type guards for narrowing
 */
export function isLifecycleEvent(event: PlaybackEvent): event is { type: 'lifecycle-changed'; state: AudioLifecycle } {
  return event.type === 'lifecycle-changed';
}

export function isPlaybackStartedEvent(event: PlaybackEvent): event is { type: 'playback-started'; url: string } {
  return event.type === 'playback-started';
}

export function isErrorEvent(event: PlaybackEvent): event is { type: 'error'; message: string } {
  return event.type === 'error';
}
```

```typescript
// PlaybackRequest schema - validates play() arguments
export const PlaybackRequestSchema = z.object({
  url: z.string().url('Invalid audio URL'),
  loop: z.boolean().optional().default(false),
  volume: z.number().min(0).max(1).optional().default(0.8),
});

export type ValidatedPlaybackRequest = z.infer<typeof PlaybackRequestSchema>;

/**
 * PlaybackEvent schema - discriminated union of audio events
 */
export const PlaybackEventSchema = z.discriminatedUnion('type', [
  z.object({
    type: z.literal('lifecycle-changed'),
    state: AudioLifecycleSchema,
  }),
  z.object({
    type: z.literal('playback-started'),
    url: z.string().url(),
  }),
  z.object({
    type: z.literal('playback-stopped'),
  }),
  z.object({
    type: z.literal('error'),
    message: z.string(),
  }),
]);

export type ValidatedPlaybackEvent = z.infer<typeof PlaybackEventSchema>;

/**
 * Validator functions - safe validation with typed results
 */
export function validatePlaybackRequest(request: unknown) {
  return PlaybackRequestSchema.safeParse(request);
}

export function validatePlaybackEvent(event: unknown) {
  return PlaybackEventSchema.safeParse(event);
}

/**
 * Runtime type checks - for use before dispatching
 */
export function isValidPlaybackRequest(request: unknown): request is ValidatedPlaybackRequest {
  return PlaybackRequestSchema.safeParse(request).success;
}

export function isValidPlaybackEvent(event: unknown): event is ValidatedPlaybackEvent {
  return PlaybackEventSchema.safeParse(event).success;
}
```
