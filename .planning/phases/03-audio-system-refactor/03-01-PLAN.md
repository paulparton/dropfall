```typescript
import type { AudioContext, AudioLifecycle, PlaybackRequest, PlaybackEvent } from '../types/Audio';
import { validatePlaybackRequest } from '../validation/schemas';

/**
 * AudioSystem - Manages Web Audio API with explicit lifecycle
 * Implements singleton pattern, initialized on first user input
 */
export class AudioSystem implements AudioContext {
  private lifecycle: AudioLifecycle = 'uninitialized';
  private audioContextHandle: Web​Audio​Context | null = null;
  private gainNode: GainNode | null = null;
  private bufferCache: Map<string, AudioBuffer> = new Map();
  private listeners: Set<(event: PlaybackEvent) => void> = new Set();

  constructor() {
    // Non-initialized state until initialize() is called
  }

  /**
   * Lifecycle property - tracks audio system state
   */
  get lifecycle_state(): AudioLifecycle {
    return this.lifecycle;
  }

  /**
   * Initialize Web Audio context on first use
   * Called automatically on first user input
   */
  async initialize(): Promise<void> {
    if (this.lifecycle !== 'uninitialized') {
      console.warn('AudioSystem already initialized');
      return;
    }

    try {
      this.lifecycle = 'initializing';
      this.emit({ type: 'lifecycle-changed', state: 'initializing' });

      // Create Web Audio context
      const AudioContext = window.AudioContext || (window as any).webkitAudioContext;
      if (!AudioContext) {
        throw new Error('Web Audio API not supported');
      }

      this.audioContextHandle = new AudioContext();
      this.gainNode = this.audioContextHandle.createGain();
      this.gainNode.connect(this.audioContextHandle.destination);

      // Set initial volume
      this.gainNode.gain.value = 0.8;

      this.lifecycle = 'ready';
      this.emit({ type: 'lifecycle-changed', state: 'ready' });
    } catch (error) {
      this.lifecycle = 'uninitialized';
      this.emit({ type: 'error', message: String(error) });
      throw error;
    }
  }

  /**plays audio from buffer
   * Play audio - must call initialize() first
   */
  async play(request: PlaybackRequest): Promise<void> {
    // Validate request
    validatePlaybackRequest(request);

    if (this.lifecycle !== 'ready' && this.lifecycle !== 'playing') {
      throw new Error(`Cannot play: AudioSystem is ${this.lifecycle}, call initialize() first`);
    }

    try {
      this.lifecycle = 'playing';
      // Load buffer if not cached
      const buffer = await this.getOrLoadBuffer(request.url);
      
      // Create source and play
      const source = this.audioContextHandle!.createBufferSource();
      source.buffer = buffer;
      source.loop = request.loop || false;
      source.connect(this.gainNode!);
      source.start(0);

      this.emit({ type: 'playback-started', url: request.url });
    } catch (error) {
      this.emit({ type: 'error', message: `Failed to play: ${error}` });
      throw error;
    }
  }

  /**
   * Stop all playback
   */
  stop(): void {
    if (this.lifecycle === 'playing') {
      this.lifecycle = 'ready';
      // Stop all sources (implementation depends on tracking)
      this.emit({ type: 'playback-stopped' });
    }
  }

  /**
   * Set master volume (0-1)
   */
  setVolume(level: number): void {
    if (!this.gainNode) return;
    this.gainNode.gain.value = Math.max(0, Math.min(1, level));
  }

  /**
   * Dispose - cleanup Web Audio resources
   */
  dispose(): void {
    try {
      this.stop();
      
      // Clear buffers
      this.bufferCache.clear();
      
      // Close audio context
      if (this.audioContextHandle) {
        this.audioContextHandle.close().catch(e => console.warn('Error closing audio context:', e));
        this.audioContextHandle = null;
      }

      this.gainNode = null;
      this.listeners.clear();
      
      this.lifecycle = 'disposed';
      this.emit({ type: 'lifecycle-changed', state: 'disposed' });
    } catch (error) {
      console.error('Error disposing AudioSystem:', error);
    }
  }

  /**
   * Private: Load or get cached buffer
   */
  private async getOrLoadBuffer(url: string): Promise<AudioBuffer> {
    if (this.bufferCache.has(url)) {
      return this.bufferCache.get(url)!;
    }

    const response = await fetch(url);
    const arrayBuffer = await response.arrayBuffer();
    const buffer = await this.audioContextHandle!.decodeAudioData(arrayBuffer);
    
    this.bufferCache.set(url, buffer);
    return buffer;
  }

  /**
   * Private: Emit events to listeners
   */
  private emit(event: PlaybackEvent): void {
    this.listeners.forEach(listener => listener(event));
  }

  /**
   * Subscribe to audio events
   */
  subscribe(listener: (event: PlaybackEvent) => void): () => void {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }
}

// Singleton instance
let instance: AudioSystem | null = null;

export function getAudioSystem(): AudioSystem {
  if (!instance) {
    instance = new AudioSystem();
  }
  return instance;
}
```
