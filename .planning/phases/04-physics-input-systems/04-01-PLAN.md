export interface CollisionEvent {

  entityA: string;
  entityB: string;
  started: boolean;
  impulse: number;
}

  type: 'knockback';
  targetEntity: string;
  force: { x: number; y: number };
  duration: number;
}

  type: 'out-of-bounds';
  entity: string;
  lastPosition: { x: number; y: number };
  direction: 'up' | 'down' | 'left' | 'right' | 'corner';
}

export type PhysicsEvent = CollisionEvent | KnockbackEvent | OutOfBoundsEvent;
