From src/types/Input.ts:
```typescript
export type InputSource = 'keyboard' | 'gamepad' | 'ai';

export interface KeyboardInput {
  source: 'keyboard';
  up: boolean;
  down: boolean;
  left: boolean;
  right: boolean;
  boost: boolean;
  timestamp: number;
}

export interface GamepadInput {
  source: 'gamepad';
  gamepadIndex: number;
  up: boolean;
  down: boolean;
  left: boolean;
  right: boolean;
  boost: boolean;
  analog: { x: number; y: number };
  timestamp: number;
}

export interface AIInput {
  source: 'ai';
  difficulty: 'easy' | 'normal' | 'hard';
  targetPosition: { x: number; y: number };
  up: boolean;
  down: boolean;
  left: boolean;
  right: boolean;
  boost: boolean;
  timestamp: number;
}

export type InputPayload = KeyboardInput | GamepadInput | AIInput;

export interface InputHandler {
  subscribe(callback: (input: InputPayload) => void): () => void;
  dispatch(input: InputPayload): void;
  getLastInput(): InputPayload | null;
  destroy(): void;
}
```
