2. Type definitions:
   ```typescript
   interface StoreActions {
     setGameMode(mode: GameMode): void;
     setDifficulty(difficulty?: Difficulty): void;
     addEntity(entity: Entity): void;
     updateEntity(id: string, updates: Partial<Entity>): void;
     removeEntity(id: string): void;
     updateScore(p1: number, p2: number): void;
     endRound(): void;
     reset(): void;
   }
   
   type GameStore = GameState & StoreActions;
   ```

3. Create Zustand store:
   ```typescript
   export const useGameStore = create<GameStore>()(
     persist(
       (set) => ({
         // Initial state matching GameState
         id: generateUUID(),
         mode: '2P' as GameMode,
         phase: 'menu' as const,
         players: [],
         arena: null,
         entities: {},
         round: 0,
         score: { p1: 0, p2: 0 },
         isOnline: false,
         createdAt: new Date(),
         updatedAt: new Date(),
         
         // Actions with validation
         setGameMode(mode) {
           set((state) => {
             const updated = { ...state, mode, updatedAt: new Date() };
             validateGameState(updated); // Will throw if invalid
             return updated;
           });
         },
         
         // ... other actions
       }),
       {
         name: 'game-store',
         // Optionally: partialize to exclude some fields from storage
       }
     )
   );
   ```

   ```typescript
   export function validateStoreMutation(state: Partial<GameState>): void {
     // Will be implemented with schema in Task 2
   }
   ```

5. Type the store hook properly:
   - Return type should be: `UseStore<GameStore>`
   - All store actions should be typed: `(args) => void` with proper signatures

**Imports:**
```typescript
import { z } from 'zod';
import type { GameState, GameMode, Difficulty } from '../types/Game';
import type { InputPayload, KeyboardInput, GamepadInput, AIInput } from '../types/Input';
import type { PhysicsEvent } from '../types/Physics';
import type { AudioContext } from '../types/Audio';

```

**Exports (required):**

1. GameModeSchema: z.enum(['1P', '2P', 'AI'])

2. DifficultySchema: z.enum(['easy', 'normal', 'hard'])

3. GamePhaseSchema: z.enum(['menu', 'loading', 'playing', 'paused', 'roundEnd', 'gameEnd'])

4. EntityMapSchema: z.record(z.string(), z.object({
     id: z.string(),
     type: z.string(),
     state: z.enum(['created', 'ready', 'active', 'destroyed']),
     position: z.object({ x: z.number(), y: z.number() }),
     // ... other Entity properties
   }))

5. GameStateSchema: z.object({
     id: z.string(),
     mode: GameModeSchema,
     phase: GamePhaseSchema,
     difficulty: DifficultySchema.optional(),
     players: z.array(z.object({ ... })).min(1).max(2),
     arena: z.object({ ... }),
     entities: EntityMapSchema,
     round: z.number().min(0),
     score: z.object({ p1: z.number().min(0), p2: z.number().min(0) }),
     isOnline: z.boolean(),
     createdAt: z.date(),
     updatedAt: z.date(),
   })

6. InputPayloadSchema: z.discriminatedUnion('source', [
     z.object({
       source: z.literal('keyboard'),
       up: z.boolean(),
       down: z.boolean(),
       left: z.boolean(),
       right: z.boolean(),
       boost: z.boolean(),
       timestamp: z.number(),
     }),
     z.object({
       source: z.literal('gamepad'),
       gamepadIndex: z.number(),
       up: z.boolean(),
       down: z.boolean(),
       left: z.boolean(),
       right: z.boolean(),
       boost: z.boolean(),
       analog: z.object({ x: z.number(), y: z.number() }),
       timestamp: z.number(),
     }),
     z.object({
       source: z.literal('ai'),
       difficulty: z.enum(['easy', 'normal', 'hard']),
       targetPosition: z.object({ x: z.number(), y: z.number() }),
       up: z.boolean(),
       down: z.boolean(),
       left: z.boolean(),
       right: z.boolean(),
       boost: z.boolean(),
       timestamp: z.number(),
     }),
   ])

7. PhysicsEventSchema: z.discriminatedUnion('type', [
     z.object({
       type: z.literal('collision'),
       entityA: z.string(),
       entityB: z.string(),
       started: z.boolean(),
       impulse: z.number(),
     }),
     // ... knockback, out-of-bounds
   ])

8. NetworkMessageSchema: z.discriminatedUnion('type', [
     z.object({
       type: z.literal('join'),
       version: z.literal('1.0'),
       playerId: z.string(),
       gameMode: z.enum(['1P', '2P']),
       playerName: z.string(),
     }),
     // ... start-round, player-move, etc.
   ])

9. AudioContextSchema: z.object({
     lifecycle: z.enum(['uninitialized', 'initializing', 'ready', 'playing', 'disposed']),
     play: z.function(),
     stop: z.function(),
     setVolume: z.function(),
     dispose: z.function(),
   })

10. Validator functions:
    ```typescript
    export function validateGameState(state: unknown): GameState {
      return GameStateSchema.parse(state);
    }
    
    export function validateInputPayload(input: unknown): InputPayload {
      return InputPayloadSchema.parse(input);
    }
    
    export function validatePhysicsEvent(event: unknown): PhysicsEvent {
      return PhysicsEventSchema.parse(event);
    }
    
    export function validateNetworkMessage(message: unknown): NetworkMessage {
      return NetworkMessageSchema.parse(message);
    }
    ```

**Rules:**
- Use z.discriminatedUnion for type discrimination (more efficient than z.union)
- Add .describe() to schemas for error clarity
- Coerce types where needed (e.g., z.date().coerce for date parsing)
- Keep schemas aligned with type definitions from Phase 2

```typescript
import { describe, it, expect } from 'vitest';
import { useGameStore } from '../src/store';
import {
  GameStateSchema,
  InputPayloadSchema,
  PhysicsEventSchema,
  NetworkMessageSchema,
  validateGameState,
  validateInputPayload,
  validatePhysicsEvent,
  validateNetworkMessage,
} from '../src/validation/schemas';
import type { GameState, GameMode } from '../src/types/Game';
import type { InputPayload } from '../src/types/Input';

describe('Type System', () => {
  describe('Store types', () => {
    it('should initialize store with valid GameState', () => {
      const store = useGameStore.getState();
      expect(store.mode).toBeDefined();
      expect(['1P', '2P', 'AI']).toContain(store.mode);
      expect(store.score).toHaveProperty('p1');
      expect(store.score).toHaveProperty('p2');
    });

    it('should accept valid GameMode updates', () => {
      useGameStore.setState({ mode: '1P' as GameMode });
      const state = useGameStore.getState();
      expect(state.mode).toBe('1P');
    });
  });

  describe('GameStateSchema validation', () => {
    it('should validate complete GameState', () => {
      const validState: GameState = {
        id: 'game-1',
        mode: '1P',
        phase: 'playing',
        difficulty: 'normal',
        players: [],
        arena: null as any, // placeholder for Entity
        entities: {},
        round: 1,
        score: { p1: 0, p2: 0 },
        isOnline: false,
        createdAt: new Date(),
        updatedAt: new Date(),
      };
      
      const result = GameStateSchema.safeParse(validState);
      expect(result.success).toBe(true);
    });

    it('should reject invalid GameState (missing id)', () => {
      const invalidState = {
        mode: '1P',
        phase: 'playing',
        // missing id, players, etc.
      };
      
      const result = GameStateSchema.safeParse(invalidState);
      expect(result.success).toBe(false);
    });

    it('should reject invalid GameMode', () => {
      const invalidState = {
        id: 'game-1',
        mode: 'invalid' as any,
        phase: 'playing',
        players: [],
        entities: {},
        round: 0,
        score: { p1: 0, p2: 0 },
        isOnline: false,
        createdAt: new Date(),
        updatedAt: new Date(),
      };
      
      const result = GameStateSchema.safeParse(invalidState);
      expect(result.success).toBe(false);
    });
  });

  describe('InputPayloadSchema validation', () => {
    it('should validate KeyboardInput', () => {
      const keyboardInput: InputPayload = {
        source: 'keyboard',
        up: false,
        down: false,
        left: false,
        right: true,
        boost: false,
        timestamp: Date.now(),
      };
      
      const result = InputPayloadSchema.safeParse(keyboardInput);
      expect(result.success).toBe(true);
    });

    it('should validate AIInput with difficulty', () => {
      const aiInput: InputPayload = {
        source: 'ai',
        difficulty: 'hard',
        targetPosition: { x: 100, y: 100 },
        up: true,
        down: false,
        left: false,
        right: false,
        boost: false,
        timestamp: Date.now(),
      };
      
      const result = InputPayloadSchema.safeParse(aiInput);
      expect(result.success).toBe(true);
    });

    it('should reject invalid InputSource', () => {
      const invalidInput = {
        source: 'unknown' as any,
        up: false,
        down: false,
        left: false,
        right: false,
        boost: false,
        timestamp: Date.now(),
      };
      
      const result = InputPayloadSchema.safeParse(invalidInput);
      expect(result.success).toBe(false);
    });
  });

  describe('Validator functions', () => {
    it('should throw on invalid GameState via validateGameState', () => {
      expect(() => {
        validateGameState({ id: 'test', mode: 'invalid', phase: 'playing' });
      }).toThrow();
    });

    it('should parse valid data via validateInputPayload', () => {
      const input = {
        source: 'keyboard',
        up: true,
        down: false,
        left: false,
        right: false,
        boost: false,
        timestamp: Date.now(),
      };
      
      const result = validateInputPayload(input);
      expect(result.source).toBe('keyboard');
      expect(result.up).toBe(true);
    });
  });

  describe('Network schema', () => {
    it('should validate NetworkMessage with version', () => {
      const message = {
        type: 'join',
        version: '1.0',
        playerId: 'p1',
        gameMode: '1P',
        playerName: 'Player 1',
      };
      
      const result = NetworkMessageSchema.safeParse(message);
      expect(result.success).toBe(true);
    });

    it('should reject message with wrong version', () => {
      const message = {
        type: 'join',
        version: '2.0' as any,
        playerId: 'p1',
        gameMode: '1P',
        playerName: 'Player 1',
      };
      
      const result = NetworkMessageSchema.safeParse(message);
      expect(result.success).toBe(false);
    });
  });
});
```
